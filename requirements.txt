import os
import logging
from telegram import Update, ReplyKeyboardMarkup
from telegram.ext import Application, CommandHandler, MessageHandler, filters, ContextTypes

# راه‌اندازی لاگ‌گیر
logging.basicConfig(format='%(asctime)s - %(name)s - %(levelname)s - %(message)s', level=logging.INFO)
logger = logging.getLogger(__name__)

# توکن ربات
BOT_TOKEN = os.getenv("BOT_TOKEN", "توکن_ربات_تو_اینجا")

# گزینه‌های منو
main_menu = [
    ["📦 خرید اشتراک ربات", "🧪 تست رایگان ربات"],
    ["🎥 ویدیو عملکرد ربات", "📞 ارتباط با پشتیبانی"],
    ["📘 راهنمای استفاده", "❓ سوالات متداول"]
]

# پیام‌ها
welcome_text = "سلام 👋\nبه پشتیبانی رسمی MaxTraderr خوش اومدی ✨\nلطفاً یکی از گزینه‌های زیر رو انتخاب کن 👇"
subscribe_text = """💳 برای خرید اشتراک ربات:\n1️⃣ ماهانه - 49$\n2️⃣ سه‌ماهه - 129$\n3️⃣ شش‌ماهه - 239$\n\nبرای پرداخت و فعال‌سازی به پشتیبانی پیام بده."""
trial_text = """🧪 تست رایگان ۷ روزه:\nیوزرنیم متاتریدرتو بفرست به پشتیبانی تا فعال بشه ✨"""
video_text = """🎥 ویدیوی عملکرد ربات:\nhttps://t.me/maxtraderr/10"""
support_text = """📞 ارتباط با پشتیبانی:\n@MaxTraderrSupport"""
guide_text = """📘 راهنمای استفاده:\n1️⃣ اکسپرت رو بریز داخل متاتریدر\n2️⃣ روی چارت M1 یا H1 بنداز\n3️⃣ خودش به‌صورت خودکار معامله می‌زنه"""
faq_text = """❓ سوالات متداول:\n\n💠 روی همه بروکرها کار می‌کنه؟\nبله، تست شده.\n\n💠 VPS لازمه؟\nپیشنهاد میشه برای پایداری."""

help_text = """🤖 راهنمای دستورات:
/start - شروع
/help - راهنما

از منوی زیر استفاده کنید 👇"""

# start
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(welcome_text, reply_markup=ReplyKeyboardMarkup(main_menu, resize_keyboard=True))

# help
async def help_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(help_text, reply_markup=ReplyKeyboardMarkup(main_menu, resize_keyboard=True))

# هندل پیام‌های کاربر
async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    text = update.message.text.strip()

    if text == "📦 خرید اشتراک ربات":
        await update.message.reply_text(subscribe_text)
    elif text == "🧪 تست رایگان ربات":
        await update.message.reply_text(trial_text)
    elif text == "🎥 ویدیو عملکرد ربات":
        await update.message.reply_text(video_text)
    elif text == "📞 ارتباط با پشتیبانی":
        await update.message.reply_text(support_text)
    elif text == "📘 راهنمای استفاده":
        await update.message.reply_text(guide_text)
    elif text == "❓ سوالات متداول":
        await update.message.reply_text(faq_text)
    else:
        await update.message.reply_text("گزینه نامعتبره. لطفاً از منوی زیر انتخاب کن 👇", reply_markup=ReplyKeyboardMarkup(main_menu, resize_keyboard=True))

# هندل خطا
async def error_handler(update, context):
    logger.error(f"Update error: {context.error}")
    if isinstance(update, Update) and update.message:
        await update.message.reply_text("خطایی رخ داد. دوباره تلاش کن.")

# main
def main():
    if not BOT_TOKEN or BOT_TOKEN == "7951072696:AAEmDDiXpnxgdTG6ZbBMTYCXgLYyeCqKlBA"
        print("❌ Bot token رو باید ست کنی")
        return

    app = Application.builder().token(BOT_TOKEN).build()

    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("help", help_command))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))
    app.add_error_handler(error_handler)

    print("🤖 MaxTraderr Support Bot is running...")
    app.run_polling()

if __name__ == "__main__":
    main()
